\begin{gather*}
\forall x\,\left(
0 \mNE x + 1
\right)
\\
\forall x\,\left(
x + 0 \mEQ x
\right)
\\
\forall x\forall y\,\left(
x + 1 \mEQ y + 1 \limp x \mEQ y
\right)
\\
\forall x\forall y\,\left(
x+(y+1) \mEQ (x+y)+1
\right)\\
\underbrace{G(0)}_{\text{base}} 
\land\, \forall x\,\underbrace{
\left(G(x) \limp G(x+1)\right)
}_{\text{step case}}
\,\limp\, \forall y\,G(y)
\tag{induction}
\end{gather*}
