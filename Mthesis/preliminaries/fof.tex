\begin{definition}\label{def:formula}
%Let $\mcF$ be a signature. 
We build the set of first order formulae with atoms, logical connectives, and logical quantifiers.
%
\[
 F ::= P(t_1,\ldots,t_n) \mid
	s\mEQ t \mid
	(\lnot F) \mid 
	(F\lor F) \mid
	(F\land F) \mid
	(F\rightarrow F) \mid
	(\forall x F) \mid
	(\exists x F)
\]
%negation $\lnot F$,
%universal quantification $\forall x F$, 
%existential quantification $\exists x F$, 
%%of one formula $\phi$;
%disjunction $ F\lor F'$, 
%conjunction $ F\land F'$, 
%or implication $ F\rightarrow F'$. 
%Exclusive disjunction $ F\oplus F$ abbreviates $( F\lor F)\land(\lnot\phi\lor\lnot F)$.
%of two formulae $\phi$ and $\psi$.
\end{definition}

\noindent Note: We abbreviate $\lnot(s\mEQ t)$ with $s\mNE t$.
