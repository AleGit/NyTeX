\subsection{Subsumption}
		\frame{
			\begin{block}{Subsumption}
				\vspace{-1em}
%				Subsumed clauses can be removed from a set of first order clauses without affecting satisfiability.
%				Proper instances of clauses affect the satisfiability of the set of ground instances.
				\begin{gather*}
					S = \{C, D, \ldots \}, \exists\theta\ C\theta\subseteq D \tag*{C subsumes D}\\
					 S \text{ satisfiable } 
					 \overset{\text{\colHi\cmark}}\iff 
					  (S \setminus D) \text{ satisfiable } \\
					S\bot \text{ satisfiable } 
					{\colLo\overset{\text{\colLo\xmark}}\iff} 
					(S \setminus D)\bot \text{ satisfiable }
					\end{gather*}
				\end{block}
			
			\begin{Example}[]
				\vspace{-1em}
					\begin{align*}
						S_0 =&\ 
						\{
							\mP(x,y), \lnot\mP(\ma,z)
						\} \\
						S_0\bot =&\ 
						\{
						\mP(\bot,\bot), \lnot\mP(\ma,\bot)\tag*{satisfiable}
						\} \\
						&\quad\infer[
						x\mapsto\ma, z\mapsto y]{
							\mP(\ma,y)\quad 
							{\colG\lnot\mP(\ma,y)}}{
							\mP(x,y) & \lnot\mP(\ma,z)}\\
						S_1 =&\ 
						\{
						\mP(x,y), \lnot\mP(\ma,z), \mP(\ma,y)
						\} \\
						S_1\bot =&\ 
						\{
						\mP(\bot,\bot), 
						{\colLo\lnot\mP(\ma,\bot)}, 
						{\colLo\mP(\ma,\bot)}
						\}\tag*{unsatisfiable}
					\end{align*}
				
				\end{Example}
