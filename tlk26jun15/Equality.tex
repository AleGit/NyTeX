\subsection{Schemata}

\begin{frame}
	\begin{block}{Schemata}
		\vspace{-1em}
		\begin{align*}		
			&x=x &s\neq s
			\\
			{\colG x\neq y\ \lor}\ 
				&y=x &s \neq t
			\\
			{\colG x\neq y \lor y\neq z\ \lor}\ 
				&x=z &s\neq t
			\\
			{\colG x_1\neq y_1 \lor x_2\neq y_2 \ \lor}\ 
			&\mf(x_1,x_2) = \mf(y_1,y_2) 
				&\mf(s)\neq\mf(t)
			\\
		    {\colG x\neq y\ \lor}\ &\lnot\mP(x) \lor \mP(y) 
			    &\mP(s), \lnot\mP(t)
			\end{align*}
		\end{block}
